Hint: Use fetch() to load JSON data, and document.createElement() to display the data.
Answer: fetch('students.json')
.then(response => response.json())
.then(data => {
  data.forEach(student => {
    let div = document.createElement('div');
    div.innerHTML = `Name: ${student.name}, Age: ${student.age}, Grade: ${student.grade}`;
    document.body.appendChild(div);
  });
});

Explanation: This JavaScript code uses fetch() to load the JSON data and iterates over the student list to display their information on the webpage using DOM manipulation.

Hint: Use document.cookie in JavaScript to store and retrieve cookie values.
Answer: <form>
  <label for="theme">Choose Theme:</label>
  <input type="text" id="theme" name="theme"><br>
  <label for="language">Choose Language:</label>
  <input type="text" id="language" name="language"><br>
  <input type="submit" value="Save Preferences">
</form>

<script>
document.querySelector('form').addEventListener('submit', function(e) {
  e.preventDefault();
  document.cookie = `theme=${document.getElementById('theme').value}; path=/`;
  document.cookie = `language=${document.getElementById('language').value}; path=/`;
});

function getCookie(name) {
  let value = "; " + document.cookie;
  let parts = value.split("; " + name + "=");
  if (parts.length === 2) return parts.pop().split(';').shift();
}

document.getElementById('theme').value = getCookie('theme') || '';
document.getElementById('language').value = getCookie('language') || '';
</script>

Explanation: This code captures user input for theme and language preferences, stores them as cookies, and retrieves the stored cookies when the user revisits the page.
